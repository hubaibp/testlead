function performClearData(exportFirst = false) {
  if(exportFirst && leadsData.length > 0) {
    window.location.href = '/api/export/csv?type=leads';
    setTimeout(() => { executeClear(); }, 800);
  } else { executeClear(); }
}
function executeClear() {
  fetch('/api/clear_data', { method: 'POST' })
  .then(res => res.json())
  .then(data => {
    if(data.status === 'cleared') {
      allData = []; leadsData = []; totalScraped = 0; totalLeads = 0;
      updateStats(0,0,0);
      renderTables();
      document.getElementById('live-tbody').innerHTML = '';
      document.getElementById('terminal').innerHTML = `<div class="log-line"><span class="log-time">${new Date().toTimeString().split(' ')[0]}</span><span class="log-msg info">🗑️ All data cleared by user. Ready for new scrape.</span></div>`;
      showToast('All leads & businesses cleared!', 'success');
      updateExportStats();
    } else { showToast('Clear failed', 'error'); }
  }).catch(()=> showToast('Error clearing data', 'error'));
  clearModal.classList.remove('active');
}
